% #################################################
% # 使用说明
% # 1、输入：向量a为超级参数，n为可能状态数，可以改变
% # 2、输出：r为分布概率
% # gamrnd: gamma函数
% #################################################


a = [1 2 1 3];
n = 3;
p = length(a);
r = gamrnd(repmat(a,n,1),1,n,p);
r = r ./ repmat(sum(r,2),1,p);
